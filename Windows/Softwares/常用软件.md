# Git
  ```Shell
  function gp()
  {
      if [[ `git status | grep modified` != "" ]]; then
        git stash && git pull --rebase && git stash pop
        if [[ $? != 0 ]]; then
          echo ''
          git status
          echo -e '\033[41;33mError!!!033[0m'
        fi
      else
        echo 'No change, use "git pull --rebase"'
        echo ''
        git pull --rebase
      fi
  }
  
  function gs()
  {
      git status $@
  }
  
  function gc()
  {
      git commit -m "$*"
  }
  ```

# 校验
  * SHA-512: `certutil -hashfile $file SHA512 | grep -o '[0-9a-f]\{128\}'`
  * MD5: `certutil -hashfile $file MD5 | grep -o '[0-9a-f]\{32\}'`
